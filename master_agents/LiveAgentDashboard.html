Live Agent Monitor

${statusHTML}
AgentStatusDescription
`; fs.writeFileSync(dashboardPath, fullHTML); logEvent('Dashboard updated with latest agent statuses.'); } // --- SIMULATE TASKS --- function runAgentTask(agentName) { const agent = agents.find(a => a.name === agentName); if (!agent) return; agent.status = 'running'; updateLiveDashboard(); logEvent(`${agentName} started.`); setTimeout(() => { agent.status = 'idle'; updateLiveDashboard(); logEvent(`${agentName} completed.`); }, 3000); // simulate 3s work } // --- FILE WATCHER --- watchPaths.forEach(dir => { fs.watch(dir, { recursive: true }, (eventType, filename) => { if (!filename.includes('node_modules') && !filename.includes('.git')) { logEvent(`Detected ${eventType} in ${filename}`); runAgentTask('syncAgent'); } }); }); // --- INITIAL BOOTSTRAP --- function bootstrapSystem() { logEvent('âœ… AutoSyncAgentSystem booted.'); agents.forEach(agent => runAgentTask(agent.name)); updateLiveDashboard(); } // --- LISTENERS --- agentSystem.on('trigger', agentName => { runAgentTask(agentName); }); // --- STARTUP --- bootstrapSystem();
